#include <bits/stdc++.h>
using namespace std;

class maxSack
{
    stack<pair<int, int>> s;

    public:
    void push(int x)
    {
        if (s.empty())
        {
            s.push({x, x});
            return;
        }

        int maxx = max(s.top().second, x);
        s.push({x, maxx});
    }

    void pop()
    {
        s.pop();
    }

    int top()
    {
        if (s.empty())
        {
            cout << "stack is empty" << endl;
            return INT_MIN;
        }

        return s.top().first;
    }

    int getMax()
    {
        if (s.empty())
        {
            cout << "stack is empty" << endl;
            return INT_MIN;
        }

        return s.top().second;
    }
};

int main()
{
    maxSack s;
    s.push(3);
    s.push(5);
    cout << s.getMax() << endl;

    s.push(7);
    s.push(19);
    cout << s.getMax() << endl;

    s.pop();
    cout << s.getMax() << endl;

    s.pop();
    cout << s.top() << endl;
}

/*

5
19
7
5

*/
