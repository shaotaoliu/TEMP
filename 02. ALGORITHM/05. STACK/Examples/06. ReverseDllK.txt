#include <bits/stdc++.h>
using namespace std;

struct Node
{
    int data;
    Node *next, *prev;

    Node(int i)
    {
        data = i;
    }
};

void printList(Node* head)
{
    while (head != NULL)
    {
        cout << head->data << " ";
        head = head->next;
    }
}

Node *reverse(Node *head, int k)
{
    if (head == NULL)
    {
        return NULL;
    }

    Node *curr = head;
    Node *prev = curr->prev;
    int count = 0;

    while (count < k && curr != NULL)
    {
        curr->prev = curr->next;
        curr->next = prev;

        prev = curr;
        curr = curr->prev;
        ++count;
    }

    prev->prev = NULL;
    head->next = NULL;

    if (curr != NULL)
    {
        curr->prev = NULL;
        Node *ret = reverse(curr, k);

        head->next = ret;
        ret->prev = head;
    }

    return prev;
}

int main()
{
    Node* n1 = new Node(1);
    Node* n2 = new Node(2);
    Node* n3 = new Node(3);
    Node* n4 = new Node(4);

    n1->prev = NULL;
    n1->next = n2;
    n2->prev = n1;
    n2->next = n3;
    n3->prev = n2;
    n3->next = n4;
    n4->prev = n3;
    n4->next = NULL;

    int k = 2;

    Node *head = n1;
    cout << "Original list: ";
    printList(head);

    head = reverse(head, k);
    cout << "\nModified list: ";
    printList(head);
    cout << endl << endl;

    n1->prev = NULL;
    n1->next = n2;
    n2->prev = n1;
    n2->next = n3;
    n3->prev = n2;
    n3->next = NULL;

    head = n1;
    cout << "Original list: ";
    printList(head);

    head = reverse(head, k);
    cout << "\nModified list: ";
    printList(head);
    cout << endl << endl;

    n1->prev = NULL;
    n1->next = n2;
    n2->prev = n1;
    n2->next = NULL;

    head = n1;
    cout << "Original list: ";
    printList(head);

    head = reverse(head, k);
    cout << "\nModified list: ";
    printList(head);
    cout << endl << endl;

    n1->prev = NULL;
    n1->next = NULL;

    head = n1;
    cout << "Original list: ";
    printList(head);

    head = reverse(head, k);
    cout << "\nModified list: ";
    printList(head);
}
