#include<bits/stdc++.h>
using namespace std;

struct Node
{
    int data;
    struct Node* next;
    Node (int data)
    {
        this->data = data;
        next = NULL;
    }
};

void print(Node *head)
{
    Node *temp = head;
    while (temp != NULL)
    {
        cout << temp->data << " ";
        temp = temp->next;
    }
    cout << endl;
}

void partition(Node *&head, int x)
{
    Node *first = head;
    Node *last = head;
    Node *p = head->next;

    while (p != NULL)
    {
        Node *next = p->next;

        if (p->data < x)
        {
            p->next = first;           // insert node at the head
            first = p;
        }
        else
        {
            last->next = p;           // insert node at the tail
            last = p;
        }

        p = next;
    }

    last->next = NULL;
    head = first;
}

int main()
{
    Node *head = new Node(3);
    head->next = new Node(5);
    head->next->next = new Node(8);
    head->next->next->next = new Node(5);
    head->next->next->next->next = new Node(10);
    head->next->next->next->next->next = new Node(2);
    head->next->next->next->next->next->next = new Node(1);

    cout << "Before partition:\n";
    print(head);

    partition(head, 5);

    cout << "\nAfter partition:\n";
    print(head);
}

/*

Before partition:
3 5 8 5 10 2 1

After partition:
3 2 1 5 8 5 10

*/
