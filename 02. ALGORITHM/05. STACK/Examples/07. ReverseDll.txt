#include <bits/stdc++.h>
using namespace std;

struct Node
{
    int data;
    Node *next, *prev;

    Node(int i)
    {
        data = i;
    }
};

void printList(Node* head)
{
    while (head != NULL)
    {
        cout << head->data << " ";
        head = head->next;
    }
}

void reverse(Node **head)
{
    if (head == NULL)
    {
        return;
    }

    Node *curr = *head;
    Node *prev = curr->prev;

    while (curr != NULL)
    {
        curr->prev = curr->next;
        curr->next = prev;

        prev = curr;
        curr = curr->prev;
    }

    *head = prev;
}

int main()
{
    Node* n1 = new Node(1);
    Node* n2 = new Node(2);
    Node* n3 = new Node(3);
    Node* n4 = new Node(4);

    n1->prev = NULL;
    n1->next = n2;
    n2->prev = n1;
    n2->next = n3;
    n3->prev = n2;
    n3->next = n4;
    n4->prev = n3;
    n4->next = NULL;

    Node *head = n1;
    cout << "Original list: ";
    printList(head);

    reverse(&head);
    cout << "\nModified list: ";
    printList(head);
    cout << endl << endl;

    n1->prev = NULL;
    n1->next = NULL;

    head = n1;
    cout << "Original list: ";
    printList(head);

    reverse(&head);
    cout << "\nModified list: ";
    printList(head);
}

/*

Original list: 1 2 3 4
Modified list: 4 3 2 1

Original list: 1
Modified list: 1

*/
