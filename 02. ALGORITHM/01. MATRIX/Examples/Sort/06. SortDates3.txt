#include <bits/stdc++.h>
using namespace std;

struct Date
{
    int day, month, year;
};

void countSort(int a[], int n, int d)
{
    int freq[10] = {0};
    for (int i = 0; i < n; ++i)
    {
        freq[(a[i] / d) % 10]++;
    }

    for (int i = 1; i < 10; ++i)
    {
        freq[i] += freq[i - 1];
    }

    int temp[n];
    for (int i = n - 1; i >= 0; --i)
    {
        int digit = (a[i] / d) % 10;
        temp[freq[digit] - 1] = a[i];
        freq[digit]--;
    }

    for (int i = 0; i < n; ++i)
    {
        a[i] = temp[i];
    }
}

void radixSort(int a[], int n)
{
    int maxx = a[0];
    for (int i = 1; i < n; ++i)
    {
        maxx = max(maxx, a[i]);
    }

    for (int d = 1; maxx / d > 0; d *= 10)
    {
        countSort(a, n, d);
    }
}

void sortDates(Date a[], int n)
{
    int dp[n];

    for (int i = 0; i < n; ++i)
    {
        dp[i] = a[i].year * 10000 + a[i].month * 100 + a[i].day;
    }

    radixSort(dp, n);

    for (int i = 0; i < n; ++i)
    {
        a[i].year = dp[i] / 10000;
        a[i].month = (dp[i] % 10000) / 100;
        a[i].day = dp[i] % 100;
    }
}

int main()
{
    Date a[] = {{20, 1, 2014}, {25, 3, 2011}, {3, 12, 1676},
                {18, 11, 1982}, {19, 4, 2015}, {9, 7, 2015}};

    int n = sizeof(a) / sizeof(a[0]);
    sortDates(a, n);

    for (int i = 0; i < n; i++)
    {
        printf("%3d%3d%5d\n", a[i].day, a[i].month, a[i].year);
    }
}
