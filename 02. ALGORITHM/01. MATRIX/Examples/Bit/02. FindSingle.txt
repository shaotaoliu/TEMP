#include <iostream>
#include <list>
#include <unordered_map>
using namespace std;

#define INT_SIZE 32

int getSingle(int arr[], int n)
{
    int result = 0;

    for (int i = 0; i < INT_SIZE; ++i)
    {
        int sum = 0;
        int x = (1 << i);

        // find the sum of bits at i-th position:
        for (int j = 0; j < n; ++j)
        {
            if (arr[j] & x)
            {
                ++sum;
            }
        }

        if (sum % 3 == 1)
        {
            result |= x;
        }
    }

    return result;
}

int main()
{
    int arr[] = {12345, 56789, 56789, 13579, 34567, 13579, 56789, 13579, 7, 1, 7, 1, 7, 12345, 1, 12345};
    int n = sizeof(arr) / sizeof(arr[0]);

    int a = getSingle(arr, n);
    cout << "The element : " << a;
    return 0;
}

34567
