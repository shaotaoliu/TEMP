#include <iostream>
#include "bits/stdc++.h"
using namespace std;

struct Point
{
    int x, y;
};

int orientation(Point A, Point B, Point C)
{
    int val = (B.y - A.y) * (C.x - B.x) - (B.x - A.x) * (C.y - B.y);

    if (val == 0)
    {
        return 0;         // collinear: slope(B, A) = slope(C, B)
    }

    if (val > 0)
    {
        return 1;         // clockwise, C is under: slope(B, A) > slope(C, B)
    }

    return 2;             // counter clockwise, C is above: slope(B, A) < slope(C, B)
}

class comparator
{
    Point p0;

    public:
    comparator(Point p)
    {
        this->p0 = p;
    }

    bool operator()(Point a, Point b)
    {
        return orientation(p0, a, b) == 1;
    }
};

void clockwiseSort(Point a[], int n)
{
    // find the left-most point
    int minY = a[0].y;
    int minIndex = 0;

    for (int i = 1; i < n; ++i)
    {
        if (minY > a[i].y || (minY == a[i].y && a[minIndex].x > a[i].x))
        {
            minY = a[i].y;
            minIndex = i;
        }
    }

    swap(a[0], a[minIndex]);
    sort(a + 1, a + n, comparator(a[0]));

    for (int i = 0; i < n; ++i)
    {
        printf("(%d, %d) ", a[i].x, a[i].y);
    }
}

int main()
{
    Point a[] = {{3, 0}, {2, 3},{2, 0}, {0, 1}, {0, 2},  {4, 3}, {5, 2}};
    int n = sizeof(a) / sizeof(a[0]);

    clockwiseSort(a, n);
}

/*

(2, 0) (0, 1) (0, 2) (2, 3) (4, 3) (5, 2) (3, 0) 

*/
