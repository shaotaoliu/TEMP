#include <iostream>
#include <stack>
#include <vector>
using namespace std;

int orientation(pair<int, int> a, pair<int, int> b, pair<int, int> c)
{
    int res = (b.second-a.second)*(c.first-b.first) -
    (c.second-b.second)*(b.first-a.first);

    if (res == 0)
    return 0;
    if (res > 0)
    return 1;
    return 2;
}

int distance2(pair<int, int> p1, pair<int, int> p2)
{
    return (p1.first-p2.first)*(p1.first-p2.first) +
    (p1.second-p2.second)*(p1.second-p2.second);
}

bool inside(vector<pair<int, int>> a, pair<int, int> p)
{
    pair<int, int> mid = {0, 0};
    int n = a.size();

    p.first *= n;
    p.second *= n;
    for (int i=0; i<n; i++)
    {
        mid.first += a[i].first;
        mid.second += a[i].second;
        a[i].first *= n;
        a[i].second *= n;
    }

    for (int i=0, j; i<n; i++)
    {
        j = (i+1)%n;
        int x1 = a[i].first, x2 = a[j].first;
        int y1 = a[i].second, y2 = a[j].second;
        int a1 = y1-y2;
        int b1 = x2-x1;
        int c1 = x1*y2-y1*x2;
        int for_mid = a1 * mid.first + b1 * mid.second + c1;
        int for_p = a1*p.first+b1*p.second+c1;
        if (for_mid*for_p < 0)
        return false;
    }

    return true;
}

vector<pair<int, int>> addPoint(vector<pair<int, int>> &vec, pair<int, int> &p)
{
    if (inside(vec, p))
    {
        return vec;
    }

    int left = 0;
    int right = 0;

    for (int i = 1; i < vec.size(); ++i)
    {
        int o = orientation(p, vec[left], vec[i]);
        if (o == 2 || (o == 0 && distance2(p, vec[left]) < distance2(p, vec[i])))
        {
            left = i;
        }
        else
        {
            o = orientation(p, vec[right], vec[i]);
            if (o == 1 || (o == 0 && distance2(p, vec[right]) < distance2(p, vec[i])))
            {
                right = i;
            }
        }
    }

    vector<pair<int, int>> ret;
    while (left != right)
    {
        ret.push_back(vec[right]);
        right = (right + 1) % vec.size();
    }

    ret.push_back(vec[right]);
    ret.push_back(p);
    return ret;
}

int main()
{
    vector<pair<int, int> > a;
    a.push_back({0, 0});
    a.push_back({3, -1});
    a.push_back({4, 5});
    a.push_back({-1, 4});
    int n = a.size();

    pair<int, int> p = {100, 100};
    a = addPoint(a, p);

    for (auto e : a)
    {
        cout << "(" << e.first << ", " << e.second << ") ";
    }

    return 0;
}
