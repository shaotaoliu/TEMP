#include <iostream>
#include "bits/stdc++.h"
using namespace std;

struct Point
{
    int x, y;
};

int orientation(Point p, Point q, Point r)
{
    int val = (q.y - p.y) * (r.x - q.x) -
    (q.x - p.x) * (r.y - q.y);

    if (val == 0) return 0;  // colinear
    return (val > 0)? 1: 2; // clock or counterclock wise
}

void convexHull(Point points[], int n)
{
    int minIndex = 0;
    for (int i = 1; i < n; ++i)
    {
        if (points[i].x < points[minIndex].x)
        {
            minIndex = i;
        }
    }

    vector<int> vec;
    vec.push_back(minIndex);

    int curr = minIndex;
    int next;

    while (true)
    {
        next = (curr + 1) % n;

        for (int i = 0; i < n; ++i)
        {
            if (i != curr && i != next && orientation(points[curr], points[next], points[i]) == 1)
            {
                next = i;
            }
        }

        if (next == minIndex)
        {
            break;
        }

        vec.push_back(next);
        curr = next;
    }

    for (int i = 0; i < vec.size(); ++i)
    {
        printf("(%d, %d) ", points[vec[i]].x, points[vec[i]].y);
    }
}

int main()
{
    Point points[] = {{0, 3}, {2, 2}, {1, 1}, {2, 1}, {3, 0}, {0, 0}, {3, 3}};
    int n = sizeof(points)/sizeof(points[0]);

    convexHull(points, n);
}

(0, 3) (0, 0) (3, 0) (3, 3)
