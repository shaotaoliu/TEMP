#include <bits/stdc++.h>
using namespace std;

int find(int a[], int n)
{
    sort(a, a+n);

    unordered_map<int, int> m;
    int count = 0;

    for (int i = 0; i < n; ++i)
    {
        m[a[i]] = 1;
        ++count;
    }

    for (int i = 2; i < n; ++i)
    {
        for (int j = 0; j < i; ++j)
        {
            if (a[i] % a[j] == 0)
            {
                int k = a[i] / a[j];

                if (m.find(k) != m.end())
                {
                    int mul = m[a[j]] * m[k];
                    m[a[i]] += mul;
                    count += mul;
                }
            }
        }
    }

    return count;
}

int main()
{
    int arr[] = { 2, 3, 4, 6 };
    int n = sizeof(arr)/sizeof(arr[0]);

    cout << find(arr, n) << endl;
    return 0;
}
