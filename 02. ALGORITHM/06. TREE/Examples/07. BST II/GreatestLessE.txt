#include <bits/stdc++.h> 
using namespace std; 
  
struct Node { 
    int data; 
    Node *left, *right; 
}; 
  
Node* newNode(int item) 
{ 
    Node* temp = new Node; 
    temp->data = item; 
    temp->left = temp->right = NULL; 
    return temp; 
} 
  
Node* insert(Node* node, int x) 
{ 
    if (node == NULL) 
        return newNode(x); 
  
    if (x < node->data) 
        node->left = insert(node->left, x); 
    else if (x > node->data) 
        node->right = insert(node->right, x); 
  
    return node; 
} 
  
int find(Node* root, int n)
{
    if (root == NULL)
    {
        return -1;
    }

    if (root->data == n)
    {
        return n;
    }

    if (root->data < n)
    {
        int res = find(root->right, n);
        if (res == -1)
        {
            return root->data;
        }

        return res;
    }

    return find(root->left, n);
}
  
int main() 
{ 
    Node* root = NULL;
    root = insert(root, 5);

    insert(root, 2); 
    insert(root, 1); 
    insert(root, 3); 
    insert(root, 12); 
    insert(root, 9); 
    insert(root, 20); 
  
    printf("%d", find(root, 10)); 
    return 0; 
}

/*

9
