#include <bits/stdc++.h>
using namespace std;

struct Node
{
    int data;
    Node *left, *right;
};

Node* newNode(int item)
{
    Node* temp = new Node;
    temp->data = item;
    temp->left = temp->right = NULL;

    return temp;
}

int insert(Node *&node, int x, int max)
{
    if (node == NULL)
    {
        node = newNode(x);
        return max;
    }

    if (node->data > x)
    {
        return insert(node->left, x, node->data);
    }

    return insert(node->right, x, max);
}

void replace(int a[], int n)
{
    Node *root = NULL;

    for (int i = n-1; i >= 0; --i)
    {
        a[i] = insert(root, a[i], -1);
    }
}

int main()
{
    int a[] = { 8, 58, 71, 18, 31, 32, 63, 92, 43, 3, 91, 93, 25, 80, 28 };
    int n = sizeof(a) / sizeof(a[0]);

    replace(a, n);

    for (int i = 0; i < n; i++)
    {
        cout << a[i] << " ";
    }
}
