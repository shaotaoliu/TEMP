#include <iostream>
#include <list>
using namespace std;

struct Node
{
    int data;
    Node *left, *right;

    Node(int d)
    {
        data = d;
        left = NULL;
        right = NULL;
    }
};

void weave(list<int> &a, list<int> &b, list<int> &c, list<list<int>> &ll)
{
    if (a.empty() && b.empty())
    {
        ll.push_back(c);
        return;
    }

    if (!a.empty())
    {
        c.push_back(a.front());
        a.pop_front();

        weave(a, b, c, ll);

        a.push_front(c.back());
        c.pop_back();
    }

    if (!b.empty())
    {
        c.push_back(b.front());
        b.pop_front();

        weave(a, b, c, ll);

        b.push_front(c.back());
        c.pop_back();
    }
}

list<list<int>> sequences(Node *root)
{
    list<list<int>> ll;
    list<int> c;

    if (root == NULL)
    {
        ll.push_back(c);
        return ll;
    }

    c.push_back(root->data);

    list<list<int>> lc = sequences(root->left);
    list<list<int>> rc = sequences(root->right);

    for (list<int> a : lc)
    {
        for (list<int> b : rc)
        {
            weave(a, b, c, ll);
        }
    }

    return ll;
}

int main()
{
    Node *n1 = new Node(1);
    Node *n2 = new Node(2);
    Node *n3 = new Node(3);
    Node *n4 = new Node(4);
    Node *n5 = new Node(5);

    n3->left = n2;
    n2->left = n1;
    n3->right = n4;
    n4->right = n5;

    list<list<int>> ll = sequences(n3);

    for (list<int> d : ll)
    {
        for (int i : d)
        {
            cout << i << " ";
        }

        cout << endl;
    }
}
