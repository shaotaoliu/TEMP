#include <bits/stdc++.h>
using namespace std;

struct Node
{
    struct Node* left, *right;
    int data;
};

struct Node* newNode(int data)
{
    struct Node* ptr = new Node;
    ptr->data = data;
    ptr->left = ptr->right = NULL;
    return ptr;
}

struct Node* insert(struct Node* root, int data)
{
    if (!root)
    root = newNode(data);
    else if (root->data > data)
    root->left = insert(root->left, data);
    else if (root->data < data)
    root->right = insert(root->right, data);
    return root;
}

int distanceFromRoot(Node* root, int x)
{
    if (root->data == x)
    {
        return 0;
    }

    if (root->data > x)
    {
        return distanceFromRoot(root->left, x) + 1;
    }

    return distanceFromRoot(root->right, x) + 1;
}

int distance(Node *root, int a, int b)
{
    if (root == NULL)
    {
        return -1;
    }

    if (root->data > a && root->data > b)
    {
        return distance(root->left, a, b);
    }

    if (root->data < a && root->data < b)
    {
        return distance(root->right, a, b);
    }

    return distanceFromRoot(root, a) + distanceFromRoot(root, b);
}

int main()
{
    struct Node* root = NULL;
    root = insert(root, 20);
    insert(root, 10);
    insert(root, 5);
    insert(root, 15);
    insert(root, 30);
    insert(root, 25);
    insert(root, 35);

    cout << distance(root, 5, 35);
}
