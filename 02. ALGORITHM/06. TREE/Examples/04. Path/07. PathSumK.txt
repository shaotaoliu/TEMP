#include <iostream>
#include <list>
#include <unordered_map>
#include <vector>
using namespace std;

struct Node
{
    int data;
    Node *left, *right;

    Node(int d)
    {
        data = d;
        left = NULL;
        right = NULL;
    }
};

void printPath(vector<int> &vec, int start)
{
    for (int i = start + 1; i < vec.size(); ++i)
    {
        printf("%d ", vec[i]);
    }

    cout << endl;
}

void findPath(Node *root, int sum, int k, unordered_map<int, vector<int>> m, vector<int> vec)
{
    if (root == NULL)
    {
        return;
    }

    vec.push_back(root->data);
    sum += root->data;

    if (m.find(sum - k) != m.end())
    {
        for (int i : m[sum - k])
        {
            printPath(vec, i);
        }
    }

    m[sum].push_back(vec.size() - 1);

    findPath(root->left, sum, k, m, vec);
    findPath(root->right, sum, k, m, vec);
}

void findPath(Node *root, int k)
{
    unordered_map<int, vector<int>> m;
    m[0].push_back(-1);

    vector<int> vec;
    findPath(root, 0, k, m, vec);
}

int main()
{
    Node *root = new Node(8);
    root->left = new Node(5);
    root->right = new Node(-3);
    root->right->right = new Node(11);
    root->left->left = new Node(3);
    root->left->right = new Node(2);
    root->left->left->left = new Node(3);
    root->left->left->right = new Node(-2);
    root->left->right->right = new Node(1);

    findPath(root, 8);
}
