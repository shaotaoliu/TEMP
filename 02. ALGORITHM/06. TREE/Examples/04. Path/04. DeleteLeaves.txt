#include<bits/stdc++.h>
#include<iostream>
using namespace std;

struct Node
{
    int data;
    struct Node *left, *right;
};

struct Node* newNode(int data)
{
    struct Node* newNode = new Node;
    newNode->data = data;
    newNode->left = newNode->right = NULL;
    return (newNode);
}

Node *deleteUtil(Node *node, int x)
{
    if (node == NULL)
    {
        return NULL;
    }

    node->left = deleteUtil(node->left, x);
    node->right = deleteUtil(node->right, x);

    if (node->left == NULL && node->right == NULL && node->data == x)
    {
        delete node;
        return NULL;
    }

    return node;
}

void inorder(Node* root)
{
    if (root == NULL)
    return;
    inorder(root->left);
    cout << root->data << " ";
    inorder(root->right);
}

int main(void)
{
    struct Node* root = newNode(10);
    root->left = newNode(3);
    root->right = newNode(10);
    root->left->left = newNode(3);
    root->left->right = newNode(1);
    root->right->right = newNode(3);
    root->right->right->left = newNode(3);
    root->right->right->right = newNode(3);

    deleteUtil(root, 3);
    cout << "Inorder: ";
    inorder(root);
    return 0;
}

/*

Inorder: 3 1 10 10

*/
