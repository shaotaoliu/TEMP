#include <bits/stdc++.h>
using namespace std;

struct Node
{
    int data;
    struct Node *left;
    struct Node *right;
};

Node* newNode(int data)
{
    Node *temp = new Node;
    temp->data = data;
    temp->left = NULL;
    temp->right = NULL;
    return temp;
}

int get_height(Node *root, int x, int &height)
{
    if (root == NULL)
    {
        return -1;
    }

    int lc = get_height(root->left, x, height);
    if (height != -1)
    {
        return height;
    }

    int lr = get_height(root->right, x, height);
    if (height != -1)
    {
        return height;
    }

    int h = max(lc, lr) + 1;
    if (root->data == x)
    {
        height = h;
    }

    return h;
}

int main()
{
    Node *root = newNode(1);
    root->left = newNode(2);
    root->right = newNode(3);
    root->left->left = newNode(4);
    root->left->right = newNode(5);
    root->right->left = newNode(6);
    root->right->right = newNode(7);
    root->left->left->left = newNode(8);
    root->left->left->right = newNode(9);

    int h = -1;
    get_height(root, 1, h);
    cout << h << endl;

    h = -1;
    get_height(root, 2, h);
    cout << h << endl;

    h = -1;
    get_height(root, 4, h);
    cout << h << endl;

    h = -1;
    get_height(root, 9, h);
    cout << h << endl;
}
