#include <bits/stdc++.h>
using namespace std;

struct Node
{
    int data;
    Node *left;
    Node *right;

    Node(int i)
    {
        data = i;
        left = NULL;
        right = NULL;
    }
};

bool checkUtil(Node* root, int post[], int &index)
{
    if (root == NULL)
    {
        return true;
    }

    if (checkUtil(root->left, post, index) == false)
    {
        return false;
    }

    if (checkUtil(root->right, post, index) == false)
    {
        return false;
    }

    printf("%d, %d\n", post[index], root->data);
    return post[index++] == root->data;
}

int getSize(Node *root)
{
    if (root == NULL)
    {
        return 0;
    }

    return getSize(root->left) + getSize(root->right) + 1;
}

bool check(Node* root, int post[], int n)
{
    int size = getSize(root);
    if (size != n)
    {
        return false;
    }

    int index = 0;
    return checkUtil(root, post, index);
}

int main()
{
    Node *root = new Node(1);
    Node *n2 = new Node(2);
    Node *n3 = new Node(3);
    Node *n4 = new Node(4);
    Node *n5 = new Node(5);
    Node *n6 = new Node(6);
    Node *n7 = new Node(7);
    Node *n8 = new Node(8);
    Node *n9 = new Node(9);

    root->left = n2;
    root->right = n3;
    n2->left = n4;
    n2->right = n5;
    n3->left = n6;
    n3->right = n7;
    n4->left = n8;
    n4->right = n9;

    int post[] = {8, 9, 4, 5, 2, 6, 7, 3, 1};

    int index = 0;
    bool b = check(root, post, index);
    cout << b << endl;
}
