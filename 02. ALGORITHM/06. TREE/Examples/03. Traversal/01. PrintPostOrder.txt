#include <bits/stdc++.h>
using namespace std;

void printPostOrder(int in[], int pre[], int n)
{
    if (n == 0)
    {
        return;
    }

    int i = 0;
    while (in[i] != pre[0])
    {
        ++i;
    }

    printPostOrder(in, pre + 1, i);
    printPostOrder(in + i + 1, pre + i + 1, n - i - 1);

    printf("%d ", pre[0]);
}

int main()
{
    int in[] = { 4, 2, 5, 1, 3, 6 };
    int pre[] = { 1, 2, 4, 5, 3, 6 };
    int n = sizeof(in) / sizeof(in[0]);

    printPostOrder(in, pre, n);
    return 0;
}

/*

4 5 2 6 3 1
