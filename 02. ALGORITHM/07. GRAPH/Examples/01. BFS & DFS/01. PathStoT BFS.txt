#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

class Graph
{
    int V;    // No. of vertices
    list<int> *adj;
    public:
    Graph(int V);
    void addEdge(int v, int w);
    bool isReachable(int s, int d);
};

Graph::Graph(int V)
{
    this->V = V;
    adj = new list<int>[V];
}

void Graph::addEdge(int v, int w)
{
    adj[v].push_back(w);
}

bool Graph::isReachable(int s, int d)
{
    if (s == d)
    {
        return true;
    }

    queue<int> q;
    q.push(s);

    int visited[V] = {0};
    visited[s] = 1;

    while (!q.empty())
    {
        int u = q.front();
        q.pop();

        for (list<int>::iterator it = adj[u].begin(); it != adj[u].end(); ++it)
        {
            if (*it == d)
            {
                return true;
            }

            if (visited[*it] == 0)
            {
                q.push(*it);
                visited[*it] = 1;
            }
        }
    }

    return false;
}

int main()
{
    Graph g(4);
    g.addEdge(0, 1);
    g.addEdge(0, 2);
    g.addEdge(1, 2);
    g.addEdge(2, 0);
    g.addEdge(2, 3);
    g.addEdge(3, 3);

    int u = 1, v = 3;
    bool b = g.isReachable(u, v);
    cout << (b ? "Yes" : "No") << endl;

    u = 3, v = 1;
    b = g.isReachable(u, v);
    cout << (b ? "Yes" : "No") << endl;

    return 0;
}
