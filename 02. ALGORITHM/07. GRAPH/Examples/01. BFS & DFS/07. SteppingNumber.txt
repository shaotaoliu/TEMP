#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

void dfs(int m, int n, int x)
{
    if (x >= n && x <= m)
    {
        cout << x << " ";
    }

    if (x == 0 || x > m)
    {
        return;
    }

    int last = x % 10;
    int prev = x * 10 + last - 1;
    int next = x * 10 + last + 1;

    if (last == 0)
    {
        dfs(m, n, next);
    }
    else if (last == 9)
    {
        dfs(m, n, prev);
    }
    else
    {
        dfs(m, n, prev);
        dfs(m, n, next);
    }
}

void find(int m, int n)
{
    for (int i = 0 ; i <= 9 ; ++i)
    {
        dfs(m, n, i);
    }
}

int main()
{
    int n = 0, m = 21;
    find(m, n);

    return 0;
}
