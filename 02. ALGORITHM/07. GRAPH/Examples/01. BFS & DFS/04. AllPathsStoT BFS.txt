#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

class Graph
{
    int V;
    list<int> *adj;
    void printUtil(int s, int d, bool visit[], int path[]);

    public:
    Graph(int V);
    void addEdge(int u, int v);
    void printAllPaths(int s, int d);
};

Graph::Graph(int V)
{
    this->V = V;
    adj = new list<int>[V];
}

void Graph::addEdge(int u, int v)
{
    adj[u].push_back(v);
}

void printv(vector<int> v)
{
    for (int i : v)
    {
        printf("%d, ", i);
    }

    printf("\n");
}

bool hasVisited(vector<int> v, int u)
{
    for (int i : v)
    {
        if (i == u)
        {
            return true;
        }
    }

    return false;
}

void Graph::printAllPaths(int s, int d)
{
    vector<int> vec;
    vec.push_back(s);

    queue<vector<int>> q;
    q.push(vec);

    while (!q.empty())
    {
        vector<int> vec = q.front();
        int u = vec[vec.size() - 1];
        q.pop();

        if (u == d)
        {
            printv(vec);
        }
        else
        {
            for (int v : adj[u])
            {
                if (hasVisited(vec, v) == false)
                {
                    vector<int> vec2(vec);
                    vec2.push_back(v);
                    q.push(vec2);
                }
            }
        }
    }
}

int main()
{
    Graph g(4);
    g.addEdge(0, 1);
    g.addEdge(0, 2);
    g.addEdge(0, 3);
    g.addEdge(2, 0);
    g.addEdge(2, 1);
    g.addEdge(1, 3);

    int s = 2, d = 3;
    cout << "Path:" << endl;
    g.printAllPaths(s, d);

    return 0;
}
