#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

class Graph
{
    int V;    // No. of vertices
    list<int> *adj;
    void DFSUtil(int v, bool visited[]);

    public:
    Graph(int V)
    {
        this->V = V;
        adj = new list<int>[V];
    }

    void addEdge(int v, int w);
    bool isSC();
    Graph getTranspose();
};

void Graph::addEdge(int v, int w)
{
    adj[v].push_back(w); // Add w to v’s list.
}

void Graph::DFSUtil(int u, bool visited[])
{
    visited[u] = true;
    for (int v : adj[u])
    {
        if (!visited[v])
        {
            DFSUtil(v, visited);
        }
    }
}

Graph Graph::getTranspose()
{
    Graph g(V);

    for (int u = 0; u < V; ++u)
    {
        for (int v : adj[u])
        {
            g.adj[v].push_back(u);
        }
    }

    return g;
}

bool Graph::isSC()
{
    bool visited[V];
    memset(visited, false, sizeof(visited));

    DFSUtil(0, visited);

    for (int i = 0; i < V; ++i)
    {
        if (visited[i] == false)
        {
            return false;
        }
    }

    Graph g2 = getTranspose();

    memset(visited, false, sizeof(visited));
    g2.DFSUtil(0, visited);

    for (int i = 0; i < V; ++i)
    {
        if (visited[i] == false)
        {
            return false;
        }
    }

    return true;
}

int main()
{
    Graph g1(5);
    g1.addEdge(0, 1);
    g1.addEdge(1, 2);
    g1.addEdge(2, 3);
    g1.addEdge(3, 0);
    g1.addEdge(2, 4);
    g1.addEdge(4, 2);
    g1.isSC()? cout << "Yes\n" : cout << "No\n";

    Graph g2(4);
    g2.addEdge(0, 1);
    g2.addEdge(1, 2);
    g2.addEdge(2, 3);
    g2.isSC()? cout << "Yes\n" : cout << "No\n";

    return 0;
}
