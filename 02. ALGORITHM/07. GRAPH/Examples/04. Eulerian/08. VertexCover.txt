#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

class Graph
{
    int V;
    list<int> *adj;
    int pickup(int degree[]);
    int getAnother(int u, int degree[]);

    public:
    Graph(int V);
    void addEdge(int v, int w);
    void printVertexCover();
};

Graph::Graph(int V)
{
    this->V = V;
    adj = new list<int>[V];
}

void Graph::addEdge(int v, int w)
{
    adj[v].push_back(w);
    adj[w].push_back(v);
}

// return a vertex whose degree = 1 if any;
// return the vertex with most degree, otherwise
int Graph::pickup(int degree[])
{
    int max = 0;
    int u = -1;

    for (int i = 0; i < V; ++i)
    {
        if (degree[i] > 0)
        {
            if (degree[i] == 1)
            {
                return i;
            }

            if (max < degree[i])
            {
                max = degree[i];
                u = i;
            }
        }
    }

    return u;
}

// get v for u where (u, v) exists
int Graph::getAnother(int u, int degree[])
{
    for (int v : adj[u])
    {
        if (degree[v] > 0)
        {
            return v;
        }
    }

    return -1;
}

void Graph::printVertexCover()
{
    int degree[V] = {0};
    for (int i = 0; i < V; ++i)
    {
        degree[i] = adj[i].size();
    }

    vector<int> res;
    while (true)
    {
        int u = pickup(degree);
        if (u == -1)
        {
            break;
        }

        if (degree[u] == 1)
        {
            int v = getAnother(u, degree);
            for (int k : adj[v])
            {
                --degree[k];
            }

            degree[u] = 0;
            degree[v] = 0;
            res.push_back(v);
        }
        else
        {
            for (int k : adj[u])
            {
                --degree[k];
            }

            degree[u] = 0;
            res.push_back(u);
        }
    }

    for (int i : res)
    {
        printf("%d, ", i);
    }
}

int main()
{
    Graph g(7);
    g.addEdge(0, 1);
    g.addEdge(1, 2);
    g.addEdge(2, 3);
    g.addEdge(3, 4);
    g.addEdge(4, 5);
    g.addEdge(5, 6);

    g.printVertexCover();
}
