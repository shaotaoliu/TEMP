#include <iostream>
#include "bits/stdc++.h"
using namespace std;

class Graph
{
    int V;
    list<int> *adj;
    int pickup();

    public:
    Graph(int V);
    void addEdge(int v, int w);
    void printVertexCover();
};

Graph::Graph(int V)
{
    this->V = V;
    adj = new list<int>[V];
}

void Graph::addEdge(int v, int w)
{
    adj[v].push_back(w);
    adj[w].push_back(v);
}

// return a vertex whose degree = 1 if any;
// return the vertex with most degree, otherwise
int Graph::pickup()
{
    int maxx = 0;
    int u = -1;

    for (int i = 0; i < V; ++i)
    {
        if (adj[i].size() == 1)
        {
            return adj[i].front();           // return the vertex on the other side
        }

        if (maxx < adj[i].size())
        {
            maxx = adj[i].size();
            u = i;
        }
    }

    return u;
}

void Graph::printVertexCover()
{
    vector<int> res;

    while (true)
    {
        int u = pickup();
        if (u == -1)
        {
            break;
        }

        for (int v : adj[u])
        {
            adj[v].remove(u);
        }

        adj[u].clear();
        res.push_back(u);
    }

    for (int i : res)
    {
        printf("%d, ", i);
    }
}

int main()
{
    Graph g(7);
    g.addEdge(0, 1);
    g.addEdge(1, 2);
    g.addEdge(2, 3);
    g.addEdge(3, 4);
    g.addEdge(4, 5);
    g.addEdge(5, 6);

    g.printVertexCover();
}

/*

1, 3, 5,
