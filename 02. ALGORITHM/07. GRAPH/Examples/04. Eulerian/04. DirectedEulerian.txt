#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

class Graph
{
    int V;
    list<int> *adj;

    public:
    Graph(int V);

    void addEdge(int v, int w)
    {
        adj[v].push_back(w);
    }

    bool isEulerianCycle();
    bool isSC();
    void DFSUtil(int v, bool visited[]);
    Graph getTranspose();
};

Graph::Graph(int V)
{
    this->V = V;
    adj = new list<int>[V];
}

void Graph::DFSUtil(int u, bool visited[])
{
    visited[u] = true;

    for (int v : adj[u])
    {
        if (!visited[v])
        {
            DFSUtil(v, visited);
        }
    }
}

Graph Graph::getTranspose()
{
    Graph g(V);

    for (int u = 0; u < V; ++u)
    {
        for (int v : adj[u])
        {
            g.adj[v].push_back(u);
        }
    }

    return g;
}

bool Graph::isSC()
{
    bool visited[V];
    memset(visited, false, sizeof(visited));

    // Find the first vertex with non-zero degree
    int n;
    for (n = 0; n < V; ++n)
    {
        if (adj[n].size() > 0)
        {
            break;
        }
    }

    DFSUtil(n, visited);

    for (int i = 0; i < V; ++i)
    {
        if (adj[i].size() > 0 && visited[i] == false)
        {
            return false;
        }
    }

    Graph gr = getTranspose();
    memset(visited, false, sizeof(visited));

    gr.DFSUtil(n, visited);

    for (int i = 0; i < V; ++i)
    {
        if (adj[i].size() > 0 && visited[i] == false)
        {
            return false;
        }
    }

    return true;
}

bool Graph::isEulerianCycle()
{
    if (isSC() == false)
    {
        return false;
    }

    // Check if in degree and out degree of every vertex is same
    int in[V] = {0};
    for (int i = 0; i < V; ++i)
    {
        for (int j : adj[i])
        {
            ++in[j];
        }
    }

    for (int i = 0; i < V; ++i)
    {
        if (adj[i].size() != in[i])
        {
            return false;
        }
    }

    return true;
}

int main()
{
    Graph g(5);
    g.addEdge(1, 0);
    g.addEdge(0, 2);
    g.addEdge(2, 1);
    g.addEdge(0, 3);
    g.addEdge(3, 4);
    g.addEdge(4, 0);

    cout << g.isEulerianCycle() << endl;
    return 0;
}
