#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

class Graph
{
    int V;
    list<int> *adj;

public:
    Graph(int V);
    void addEdge(int u, int v, int w);
    int shortestPath(int s, int d);
    int printShortestPath(int parent[], int v);
};

Graph::Graph(int V)
{
    this->V = V;
    adj = new list<int>[V * 2];
}

void Graph::addEdge(int u, int v, int w)
{
    if (w == 2)
    {
        adj[u].push_back(u + V);
        adj[u + V].push_back(v);
    }
    else
    {
        adj[u].push_back(v);
    }
}

int Graph::printShortestPath(int pre[], int v)
{
    if (pre[v] == -1)
    {
        printf("%d", v);
        return 0;
    }

    int d = printShortestPath(pre, pre[v]);
    if (v < V)
    {
        printf(" -> %d", v);
    }

    return d + 1;
}

int Graph::shortestPath(int src, int dest)
{
    int pre[V * 2];
    memset(pre, -1, sizeof(pre));

    queue<int> q;
    q.push(src);

    while (!q.empty())
    {
        int u = q.front();
        q.pop();

        if (u == dest)
        {
            return printShortestPath(pre, u);
        }

        for (int v : adj[u])
        {
            if (v != src && pre[v] == -1)
            {
                q.push(v);
                pre[v] = u;
            }
        }
    }

    return INT_MAX;
}

int main()
{
    // Create a graph given in the above diagram
    int V = 4;
    Graph g(V);
    g.addEdge(0, 1, 2);
    g.addEdge(0, 2, 2);
    g.addEdge(1, 2, 1);
    g.addEdge(1, 3, 1);
    g.addEdge(2, 0, 1);
    g.addEdge(2, 3, 2);
    g.addEdge(3, 3, 2);

    cout << "Shortest path: ";
    int d = g.shortestPath(0, 3);
    cout << endl << "Shortest distance: " << d << endl;

    return 0;
}

/*

Shortest path: 0 -> 1 -> 3
Shortest distance: 3
