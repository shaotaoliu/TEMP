#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

class Graph
{
    int V;
    list<pair<int, int>> *adj;

    public:
    Graph(int V);
    void addEdge(int u, int v, int w);
    int shortestPath(int s, int d);
    void printShortestPath(int parent[], int v);
};

Graph::Graph(int V)
{
    this->V = V;
    adj = new list<pair<int, int>>[V];
}

void Graph::addEdge(int u, int v, int w)
{
    adj[u].push_back({v, w});
}

void Graph::printShortestPath(int pre[], int v)
{
    if (pre[v] == -1)
    {
        printf("%d", v);
        return;
    }

    printShortestPath(pre, pre[v]);
    printf(" -> %d", v);
}

int Graph::shortestPath(int src, int dest)
{
    int pre[V];
    memset(pre, -1, sizeof(pre));

    queue<pair<int, int>> q;
    q.push({src, 0});

    while (!q.empty())
    {
        int u = q.front().first;
        int k = q.front().second;
        q.pop();

        if (u == dest)
        {
            printShortestPath(pre, u);
            return k;
        }

        for (auto it = adj[u].begin(); it != adj[u].end(); ++it)
        {
            if (it->first != src && pre[it->first] == -1)
            {
                if (it->second == 2)
                {
                    q.push({u, k + 1});
                    it->second = 1;
                }
                else
                {
                    q.push({it->first, k + 1});
                    pre[it->first] = u;
                }
            }
        }
    }

    return INT_MAX;
}

int main()
{
    // Create a graph given in the above diagram
    int V = 4;
    Graph g(V);
    g.addEdge(0, 1, 2);
    g.addEdge(0, 2, 2);
    g.addEdge(1, 2, 1);
    g.addEdge(1, 3, 1);
    g.addEdge(2, 0, 1);
    g.addEdge(2, 3, 2);
    g.addEdge(3, 3, 2);

    cout << "Shortest path: ";
    int d = g.shortestPath(0, 3);
    cout << endl << "Shortest distance: " << d << endl;

    return 0;
}

/*

Shortest path: 0 -> 1 -> 3
Shortest distance: 3
