#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

#define M 5
#define N 5

int dx[] = { -1, 0, 1, 0};
int dy[] = { 0, 1, 0, -1 };

void findDistance(char m[][N])
{
    queue<pair<pair<int, int>, int>> q;

    for (int i = 0; i < N; ++i)
    {
        for (int j = 0; j < N; ++j)
        {
            if (m[i][j] == 'G')
            {
                q.push({{i, j}, 0});
            }
        }
    }

    while (!q.empty())
    {
        int x = q.front().first.first;
        int y = q.front().first.second;
        int d = q.front().second;
        q.pop();

        for (int i = 0; i < 4; ++i)
        {
            int x2 = x + dx[i];
            int y2 = y + dy[i];

            if (x2 >= 0 && x2 < N && y2 >= 0 && y2 < N && m[x2][y2] == 'O')
            {
                m[x2][y2] = '0' + d + 1;
                q.push({{x2, y2}, d + 1});
            }
        }
    }
}

int main()
{
    char matrix[][N] =
    {{'O', 'O', 'O', 'O', 'G'},
     {'O', 'W', 'W', 'O', 'O'},
     {'O', 'O', 'O', 'W', 'O'},
     {'G', 'W', 'W', 'W', 'O'},
     {'O', 'O', 'O', 'O', 'G'}};

    findDistance(matrix);

    for (int i = 0; i < N; ++i)
    {
        for (int j = 0; j < N; ++j)
        {
            printf("%c  ", matrix[i][j]);
        }
        cout << endl;
    }

    return 0;
}
