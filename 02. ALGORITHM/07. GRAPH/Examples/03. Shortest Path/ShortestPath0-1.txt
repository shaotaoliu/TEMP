#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

#define V 9

void enqueue(vector<pair<int, int>> edges[], queue<pair<int, int>> &q, int dist[], int s, int k)
{
    q.push({s, k});
    dist[s] = k;

    for (auto p : edges[s])
    {
        if (dist[p.first] == -1 && p.second == 0)
        {
            enqueue(edges, q, dist, p.first, k);
        }
    }
}

void BFS(vector<pair<int, int>> edges[], int src)
{
    int dist[V];
    memset(dist, -1, sizeof(dist));

    queue<pair<int, int>> q;
    enqueue(edges, q, dist, src, 0);

    while (!q.empty())
    {
        int u = q.front().first;
        int k = q.front().second;
        q.pop();

        for (auto p : edges[u])
        {
            if (dist[p.first] == -1)
            {
                enqueue(edges, q, dist, p.first, k + 1);
            }
        }
    }

    for (int i = 0; i < V; ++i)
    {
        printf("%d ", dist[i]);
    }
}

void add_edge(vector<pair<int, int>> edges[], int u, int v, int wt)
{
    edges[u].push_back({v, wt});
    edges[v].push_back({u, wt});
}

int main()
{
    vector<pair<int, int>> edges[V];
    add_edge(edges, 0, 1, 0);
    add_edge(edges, 0, 7, 1);
    add_edge(edges, 1, 7, 1);
    add_edge(edges, 1, 2, 1);
    add_edge(edges, 2, 3, 0);
    add_edge(edges, 2, 5, 0);
    add_edge(edges, 2, 8, 1);
    add_edge(edges, 3, 4, 1);
    add_edge(edges, 3, 5, 1);
    add_edge(edges, 4, 5, 1);
    add_edge(edges, 5, 6, 1);
    add_edge(edges, 6, 7, 1);
    add_edge(edges, 7, 8, 1);

    int src = 0;
    BFS(edges, src);
    return 0;
}
