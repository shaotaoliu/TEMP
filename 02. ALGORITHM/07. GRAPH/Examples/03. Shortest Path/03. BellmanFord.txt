#include <iostream>
#include <queue>
#include <bits/stdc++.h> 
#include <map>
#include <stack>
using namespace std;
  
struct Edge
{
    int src;
    int dest;
    int weight;
};

class Graph
{
    int V;
    vector<Edge> edges;

public:
    Graph(int n)
    {
        V = n;
    }

    void add_edge(int u, int v, int weight)
    {
        edges.push_back({u, v, weight});
    }

    void BellmanFord(int src);
};

void Graph::BellmanFord(int src)
{
    int dist[V];
    int path[V];

    for (int i = 0; i < V; ++i)
    {
        dist[i] = INT_MAX;
        path[i] = -1;
    }

    dist[src] = 0;

    // relax edges repeatedly
    for (int i = 1; i < V; ++i)
    {
        for (vector<Edge>::iterator it = edges.begin(); it != edges.end(); ++it)
        {
            if (dist[it->src] != INT_MAX && dist[it->src] + it->weight < dist[it->dest])
            {
                dist[it->dest] = dist[it->src] + it->weight;
                path[it->dest] = it->src;
            }
        }
    }

    // check for negative-weight cycles
    for (vector<Edge>::iterator it = edges.begin(); it != edges.end(); ++it)
    {
        if (dist[it->src] != INT_MAX && dist[it->src] + it->weight < dist[it->dest])
        {
            printf("Graph contains negative weight cycle");
            return;
        }
    }

    for (int i = 0; i < V; ++i)
    {
        printf("%d: %3d, %3d\n", i, dist[i], path[i]);
    }
}

int main()
{
    Graph g(5);

    g.add_edge(0, 1, -1);
    g.add_edge(0, 2, 4);
    g.add_edge(1, 2, 3);
    g.add_edge(1, 3, 2);
    g.add_edge(1, 4, 2);
    g.add_edge(3, 2, 5);
    g.add_edge(3, 1, 1);
    g.add_edge(4, 3, -3);

    g.BellmanFord(0);
}

/*

0:   0,  -1
1:  -1,   0
2:   2,   1
3:  -2,   4
4:   1,   1

*/
