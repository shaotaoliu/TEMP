#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

class Graph
{
    int V;
    list<int> *adj;

    public:
    Graph(int V);
    void addEdge(int v, int w);
    bool isCycle();
};

Graph::Graph(int V)
{
    this->V = V;
    adj = new list<int>[V];
}

void Graph::addEdge(int v, int w)
{
    adj[v].push_back(w); // Add w to v’s list.
}

bool Graph::isCycle()
{
    // find the in-degrees of all vertices
    int in_degree[V] = {0};
    for (int i = 0; i < V; ++i)
    {
        for (list<int>::iterator it = adj[i].begin(); it != adj[i].end(); ++it)
        {
            ++in_degree[*it];
        }
    }

    // push all vertices whose in-degree = 0 to queue
    queue<int> q;
    for (int i = 0; i < V; ++i)
    {
        if (in_degree[i] == 0)
        {
            q.push(i);
        }
    }

    int count = 0;
    while (!q.empty())
    {
        int u = q.front();
        q.pop();
        ++count;

        for (list<int>::iterator it = adj[u].begin(); it != adj[u].end(); ++it)
        {
            --in_degree[*it];
            if (in_degree[*it] == 0)
            {
                q.push(*it);
            }
        }
    }

    return count != V;
}

int main()
{
    Graph g(4);
    g.addEdge(0, 1);
    g.addEdge(0, 2);
    g.addEdge(1, 2);
    g.addEdge(2, 0);
    g.addEdge(2, 3);
    g.addEdge(3, 3);

    bool b = g.isCycle();
    cout << (b ? "Yes" : "No") << endl;
}
