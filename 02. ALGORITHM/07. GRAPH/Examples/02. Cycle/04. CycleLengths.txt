#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

class Graph
{
    int V;
    list<int> *adj;           // also vector is ok
    void find(bool visited[], int start, int u, int k, int parent);

public:
    Graph(int n)
    {
        V = n;
        adj = new list<int>[n];
    }

    void add_edge(int u, int v)
    {
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    void find_cycles(int start);
};

void Graph::find(bool visited[], int start, int u, int k, int parent)
{
    visited[u] = true;

    for (auto it = adj[u].begin(); it != adj[u].end(); ++it)
    {
        if (visited[*it] == false)
        {
            find(visited, start, *it, k + 1, u);
        }
        else if (*it != parent && *it == start)
        {
            printf("%d\n", k);
        }
    }

    visited[u] = false;
}

void Graph::find_cycles(int start)
{
    bool visited[V];
    for (int u = 0; u < V; ++u)
    {
        visited[u] = false;
    }

    int parent = -1;
    find(visited, start, start, 1, parent);
}

int main()
{
    int n = 9;
    Graph g(n);

    g.add_edge(0, 1);
    g.add_edge(0, 3);
    g.add_edge(1, 2);
    g.add_edge(1, 3);
    g.add_edge(2, 3);
    g.add_edge(2, 4);
    g.add_edge(3, 5);
    g.add_edge(4, 5);

    g.find_cycles(0);
}
