#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

class Graph
{
    int V;    // No. of vertices'
    list<int> *adj;

public:
    Graph(int V);   // Constructor
    void addEdge(int v, int w);
    void topologicalSort();
};

Graph::Graph(int V)
{
    this->V = V;
    adj = new list<int>[V];
}

void Graph::addEdge(int v, int w)
{
    adj[v].push_back(w); // Add w to v’s list.
}

void Graph::topologicalSort()
{
    int indegree[V] = {0};
    for (int i = 0; i < V; ++i)
    {
        for (auto j : adj[i])
        {
            ++indegree[j];
        }
    }

    queue<int> q;
    for (int i = 0; i < V; ++i)
    {
        if (indegree[i] == 0)
        {
            q.push(i);
        }
    }

    while (!q.empty())
    {
        int i = q.front();
        q.pop();
        printf("%d ", i);

        for (auto j : adj[i])
        {
            --indegree[j];
            if (indegree[j] == 0)
            {
                q.push(j);
            }
        }
    }
}

int main()
{
    Graph g(6);
    g.addEdge(5, 2);
    g.addEdge(5, 0);
    g.addEdge(4, 0);
    g.addEdge(4, 1);
    g.addEdge(2, 3);
    g.addEdge(3, 1);

    cout << "Following is a Topological Sort: \n";
    g.topologicalSort();

    return 0;
}
