#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

class Graph
{
    int V;
    list<int> *adj;
    bool hasCycle(int s, vector<bool> &visited, int parent);

public:
    Graph(int V);
    void addEdge(int v, int w);
    bool isCyclic();
};

Graph::Graph(int V)
{
    this->V = V;
    adj = new list<int>[V];
}

void Graph::addEdge(int v, int w)
{
    adj[v].push_back(w); // Add w to v’s list.
    adj[w].push_back(v); // Add v to w’s list.
}

bool Graph::hasCycle(int s, vector<bool> &visited, int parent)
{
    visited[s] = true;

    for (list<int>::iterator it = adj[s].begin(); it != adj[s].end(); ++it)
    {
        if (!visited[*it])
        {
            if (hasCycle(*it, visited, s))
            {
                return true;
            }
        }
        else if (*it != parent)         // If an adjacent is visited and not parent
        {
            return true;
        }
    }

    return false;
}

bool Graph::isCyclic()
{
    vector<bool> visited(V, false);

    for (int i = 0; i < V; ++i)
    {
        if (!visited[i])
        {
            if (hasCycle(i, visited, -1))
            {
                return true;
            }
        }
    }

    return false;
}

int main()
{
    Graph g1(5);
    g1.addEdge(1, 0);
    g1.addEdge(0, 2);
    g1.addEdge(2, 1);
    g1.addEdge(0, 3);
    g1.addEdge(3, 4);
    bool b = g1.isCyclic();
    cout << (b ? "Yes" : "No") << endl;

    Graph g2(3);
    g2.addEdge(0, 1);
    g2.addEdge(1, 2);
    b = g2.isCyclic();
    cout << (b ? "Yes" : "No") << endl;

    return 0;
}
