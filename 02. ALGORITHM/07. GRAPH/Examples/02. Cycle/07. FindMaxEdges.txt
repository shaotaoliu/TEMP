#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

class Graph
{
    int V;
    list<int> *adj;

public:
    Graph(int V);
    void addEdge(int v, int w);
    void findMax();
};

Graph::Graph(int V)
{
    this->V = V;
    adj = new list<int>[V];
}

void Graph::addEdge(int v, int w)
{
    adj[v].push_back(w);
}

void Graph::findMax()
{
    int indegree[V] = {0};
    for (int i = 0; i < V; ++i)
    {
        for (int j : adj[i])
        {
            indegree[j]++;
        }
    }

    queue<int> q;
    for (int i = 0; i < V; ++i)
    {
        if (indegree[i] == 0)
        {
            q.push(i);
        }
    }

    bool visited[V];
    memset(visited, false, sizeof(visited));

    while (!q.empty())
    {
        int i = q.front();
        q.pop();
        visited[i] = true;

        int pointo[V] = {0};
        for (int j : adj[i])
        {
            pointo[j] = 1;
            --indegree[j];

            if (indegree[j] == 0)
            {
                q.push(j);
            }
        }

        for (int k = 0; k < V; ++k)
        {
            if (visited[k] == false && pointo[k] == 0)
            {
                printf("%d - %d\n", i, k);
            }
        }
    }
}

int main()
{
    // Create a graph given in the above diagram
    Graph g(6);
    g.addEdge(5, 2);
    g.addEdge(5, 0);
    g.addEdge(4, 0);
    g.addEdge(4, 1);
    g.addEdge(2, 3);
    g.addEdge(3, 1);

    g.findMax();
    return 0;
}
