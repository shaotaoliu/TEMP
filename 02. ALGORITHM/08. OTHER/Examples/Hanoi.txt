#include <iostream>
using namespace std;

void hanoi(int n, char a, char b, char c)
{
    if (n == 1)
    {
        printf("Move disk 1 from %c to %c\n", a, b);
        return;
    }

    hanoi(n - 1, a, c, b);
    printf("Move disk %d from %c to %c\n", n, a, b);
    hanoi(n - 1, c, b, a);
}

int main()
{
    hanoi(4, 'A', 'B', 'C');
}
