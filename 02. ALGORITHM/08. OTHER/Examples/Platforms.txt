#include <bits/stdc++.h>
using namespace std;

int solve(int arrive[], int depart[], int n)
{
    sort(arrive, arrive + n);
    sort(depart, depart + n);

    int platforms = 1;
    int maxx = 1;
    int i = 1;
    int j = 0;

    while (i < n && j < n)
    {
        if (arrive[i] <= depart[j])
        {
            ++platforms;
            ++i;

            if (maxx < platforms)
            {
                maxx = platforms;
            }
        }
        else
        {
            --platforms;
            ++j;
        }
    }

    return maxx;
}

int main()
{
    int arr[] = {900, 940, 950, 1100, 1500, 1800};
    int dep[] = {910, 1200, 1120, 1130, 1900, 2000};
    int n = sizeof(arr) / sizeof(arr[0]);

    cout << solve(arr, dep, n);
}
