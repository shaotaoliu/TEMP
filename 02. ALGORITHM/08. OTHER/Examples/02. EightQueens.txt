#include <iostream>
#include <queue>
#include <bits/stdc++.h>
#include <map>
#include <stack>
using namespace std;

#define N 8

bool valid(int a[][N], int row, int col)
{
    for (int i = 0; i < row; ++i)
    {
        if (a[i][col] == 1)
        {
            return false;
        }
    }

    for (int i = row - 1, j = col - 1; i >= 0, j >= 0; --i, --j)
    {
        if (a[i][j] == 1)
        {
            return false;
        }
    }

    for (int i = row - 1, j = col + 1; i >= 0, j < N; --i, ++j)
    {
        if (a[i][j] == 1)
        {
            return false;
        }
    }

    return true;
}

bool solveUtil(int a[][N], int row)
{
    if (row == N)
    {
        return true;
    }

    for (int col = 0; col < N; ++col)
    {
        if (valid(a, row, col))
        {
            a[row][col] = 1;

            if (solveUtil(a, row + 1))
            {
                return true;
            }

            a[row][col] = 0;
        }
    }

    return false;
}

int main()
{
    int a[N][N] = {0};
    solveUtil(a, 0);

    for (int i = 0; i < N; ++i)
    {
        for (int j = 0; j < N; ++j)
        {
            printf("%d ", a[i][j]);
        }
        printf("\n");
    }
}
