#include<bits/stdc++.h>
using namespace std;

int sumoflength(int a[], int n)
{
    unordered_set<int> s;
    int sum = 0;
    int j = 0;

    for (int i = 0; i < n; ++i)
    {
        while (j < i && s.find(a[i]) != s.end())
        {
            s.erase(a[j]);
            ++j;
        }

        sum += (i - j + 1) * (i - j + 2) / 2;
        s.insert(a[i]);
    }

    return sum;
}

int main()
{
    int arr[] = {1, 2, 3, 4, 2, 1, 4, 1};
    int n = sizeof(arr)/sizeof(arr[0]);
    cout << sumoflength(arr, n) << endl;
    return 0;
}
