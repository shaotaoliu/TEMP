#include <bits/stdc++.h>
using namespace std;

bool checkReverse(int a[], int n)
{
    int i = 0;
    while (i < n-1 && a[i] < a[i+1])
    {
        ++i;
    }

    if (i == n-1) return true;

    int x = i == 0 ? INT_MIN : a[i-1];
    int y = a[i];

    ++i;
    while (i < n && a[i] < a[i-1])
    {
        if (a[i] < x)
        {
            return false;
        }

        ++i;
    }

    if (i == n) return true;

    while (i < n && a[i] > a[i-1])
    {
        if (a[i] < y)
        {
            return false;
        }
        ++i;
    }

    return i == n;
}

void test1()
{
    int a[] = {1, 2, 3, 4, 5, 6};
    int n = sizeof(a) / sizeof(a[0]);
    checkReverse(a, n) ? cout << "Yes" << endl : cout << "No" << endl;
}
