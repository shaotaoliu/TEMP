#include <bits/stdc++.h>
using namespace std;

void sortByOneSwap(int a[], int n)
{
    for (int i = 0; i < n-1; ++i)
    {
        if (a[i] > a[i+1])
        {
            int j = i + 1;
            while (j < n && a[j] < a[i])
            {
                ++j;
            }

            swap(a[i], a[j-1]);
            break;
        }
    }
}

void printArray(int a[], int n)
{
    for (int i = 0; i < n; i++)
    {
        cout << a[i] << ", ";
    }

    cout << endl;
}

void test1()
{
    int arr[] = {1, 2, 3, 4, 5, 6, 7, 8};
    int n = sizeof(arr)/sizeof(arr[0]);

    sortByOneSwap(arr, n);
    printArray(arr, n);
}
