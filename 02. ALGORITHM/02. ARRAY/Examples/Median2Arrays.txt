#include<bits/stdc++.h>
using namespace std;

int median(int a[], int n)
{
    if (n % 2 == 0)
    {
        return (a[n/2] + a[n/2 - 1]) / 2;
    }

    return a[n/2];
}

int getMedian(int a[], int b[], int n)
{
    if (n <= 0)
    {
        return -1;
    }

    if (n == 1)
    {
        return (a[0] + b[0]) / 2;
    }

    if (n == 2)
    {
        return (max(a[0], b[0]) + min(a[1], b[1])) / 2;
    }

    int m1 = median(a, n);
    int m2 = median(b, n);

    if (m1 == m2)
    {
        return m1;
    }

    // if m1 < m2, then median must exist in a[m1....] and b[....m2]
    if (m1 < m2)
    {
        if (n % 2 == 0)
        {
            return getMedian(a + n/2 - 1, b, n - n/2 + 1);
        }
        else
        {
            return getMedian(a + n/2, b, n - n/2);
        }
    }

    // if m1 > m2, then median must exist in a[....m1] and b[m2...]
    if (n % 2 == 0)
    {
        return getMedian(b + n/2 - 1, a, n - n/2 + 1);
    }
    else
    {
        return getMedian(b + n/2, a, n - n/2);
    }
}

int main()
{
    int a[] = {1, 12, 15, 26, 38};
    int b[] = {2, 13, 17, 30, 45};

    int n1 = sizeof(a) / sizeof(a[0]);
    int n2 = sizeof(b) / sizeof(b[0]);

    cout << getMedian(a, b, n1) ;
}
