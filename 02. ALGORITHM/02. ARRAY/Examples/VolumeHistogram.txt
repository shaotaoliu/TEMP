#include <iostream>
using namespace std;

int getVolume(int a[], int n)
{
    int left[n];
    left[0] = a[0];

    for (int i = 1; i < n; ++i)
    {
        left[i] = max(left[i-1], a[i]);
    }

    int right[n];
    right[n-1] = a[n-1];

    for (int i = n-2; i >= 0; --i)
    {
        right[i] = max(right[i+1], a[i]);
    }

    int volume = 0;
    for (int i = 0; i < n; ++i)
    {
        volume += min(left[i], right[i]) - a[i];
    }

    return volume;
}

int main()
{
    int a[] = {0, 0, 4, 0, 0, 6, 0, 0, 3, 0, 5, 0, 1, 0, 0, 0};
    int n = sizeof(a) / sizeof(a[0]);

    cout << getVolume(a, n);
}

// 26
