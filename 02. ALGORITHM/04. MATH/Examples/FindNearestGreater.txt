#include <iostream>
#include "bits/stdc++.h"
using namespace std;

int toInt(int a[], int curr)
{
    for (int i = 0; i < 10; ++i)
    {
        while (a[i] > 0)
        {
            curr = curr * 10 + i;
            a[i]--;
        }
    }

    return curr;
}

int checkString(int a[], string b, int pre)
{
    if (b.length() == 0)
    {
        return pre;
    }

    int curr = b[0] - '0';

    if (a[curr] > 0)
    {
        a[curr]--;
        int result = checkString(a, b.substr(1), pre * 10 + curr);

        if (result > -1)
        {
            return result;
        }

        a[curr]++;                      // roll back
    }

    for (int i = curr + 1; i < 10; ++i)
    {
        if (a[i] > 0)
        {
            a[i]--;
            return toInt(a, pre * 10 + i);         // move i to curr position
        }
    }

    return -1;
}

int findNearestGreater(int a, int b)
{
    string s1 = to_string(a);
    string s2 = to_string(b);

    int dp[10] = {0};
    for (char c : s1)
    {
        dp[c - '0']++;
    }

    if (s1.length() < s2.length())
    {
        return -1;
    }

    if (s1.length() > s2.length())
    {
        return toInt(dp, 0);
    }

    return checkString(dp, s2, 0);
}

int main()
{
    int a = 123456789;
    int b = 444444444;

    cout << findNearestGreater(a, b) << endl;
}
