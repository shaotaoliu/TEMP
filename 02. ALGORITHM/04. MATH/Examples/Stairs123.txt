#include <iostream>
#include <vector>
using namespace std;

int find(vector<int> &vec, int n)
{
    if (n < 0)
    {
        return 0;
    }

    if (n == 0)
    {
        for (int i : vec)
        {
            cout << i << ", ";
        }

        cout << endl;
        return 1;
    }

    int k = 0;

    for (int i = 1; i <= 3; ++i)
    {
        vec.push_back(i);
        k += find(vec, n - i);
        vec.pop_back();
    }

    return k;
}

int main()
{
    vector<int> v;
    cout << find(v, 4);
}
