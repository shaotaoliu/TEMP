#include <iostream>
#include <string>
using namespace std;

int ctoi(char c)
{
    return c >= 'A' && c <= 'D' ? (c + 10 - 'A') : (c - '0');
}

char itoc(int x)
{
    return x < 10 ? (x + '0') : (x - 10 + 'A');
}

string addBase14(string a, string b)
{
    string res = "";
    int i = a.length() - 1;
    int j = a.length() - 1;

    int carry = 0;
    for (; i >= 0 && j >= 0; --i, --j)
    {
        int sum = ctoi(a[i]) + ctoi(b[i]) + carry;
        res = itoc(sum % 14) + res;
        carry = sum / 14;
    }

    for (; i >= 0; --i)
    {
        int sum = ctoi(a[i]) + carry;
        res = itoc(sum % 14) + res;
        carry = sum / 14;
    }

    for (; j >= 0; --j)
    {
        int sum = ctoi(b[i]) + carry;
        res = itoc(sum % 14) + res;
        carry = sum / 14;
    }

    if (carry > 0)
    {
        res = "1" + res;
    }

    return res;
}

int main()
{
    string num1 = "DC2";
    string num2 = "0A3";

    cout << addBase14(num1, num2) << endl;
}
